# CommandLine uses
    1. cat
        used to type out a file (or combine files)
    2. head
        used to show the first few lines of a file
    3. tail
        used to show the last few lines of a file 
    4. man
        used to view documentation



# Structure of a command
    ls -a /home/angela

    here 'ls' is command & '-a' is -options & '/home/angela' is arguments


# Switching between GUI and CUI










# halt or poweroff cmd is 
    shutdown -h

# restart 
    shutdown -r

# which for finding the path to that 
    which man

# where for finding the whole path
    whereis brave




# accessing the directory
    pwd         -   Displays the present directory
    cd ~ or cd  -   changes to the home directory
    cd ..       -   change to the parent directory
    cd -        -   change to the previous directory
